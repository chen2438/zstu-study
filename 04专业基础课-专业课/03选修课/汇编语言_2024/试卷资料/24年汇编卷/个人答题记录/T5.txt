; 定义宏 MIN
MIN MACRO a, b, c
    ; 比较 a 和 b
    mov al, a
    cmp al, b
    jle code1
    mov al, b
code1:
    ; 比较 AL 和 c
    cmp al, c
    jle code2
    mov al, c
code2:
ENDM



data segment
    x db 1   ; +1
    y db 0   ; 0
    z db -1  ; -1
    minValue db 0

code segment
main proc
    mov ax, @data
    mov ds, ax
    
    ; 调用宏 MIN 计算 x, y, z 的最小值
    MIN x, y, z

    ; 将最小值存入 minValue
    mov minValue, al

    ; 结束程序
    mov ah,00H
    int 21h
main endp

end main
